# Iraq Events
###########################

add_namespace = iraq


#Surrender when Israel has taken too much land in Iraq
country_event = {
	id = iraq.1
	title = iraq.1.t
	desc = iraq.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	trigger = {
		tag = IRQ
		has_war_with = ISR
		surrender_progress > 0.25
		has_civil_war = no
		ISR = { has_completed_focus = ISR_pre_emptive_strike }
	}

	mean_time_to_happen = { days = 1 }

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = iraq.1.a

		ISR = {
			country_event = { id = iraq.2 }
		}
		custom_effect_tooltip = iraq.1.a_tt
		effect_tooltip = {
			ISR = { 
				white_peace = {
					tag = IRQ 
					message = IRQ_agree_peace 
				} 
			}

		}
	}
}

#Israel response to Iraq surrender
country_event = {
	id = iraq.2
	title = iraq.2.t
	desc = iraq.2.d
	picture = GFX_report_event_finnish_soldiers_snow
	
	is_triggered_only = yes

	#Accept surrender
	option = {
		name = iraq.2.a

		white_peace = IRQ
	}
}
