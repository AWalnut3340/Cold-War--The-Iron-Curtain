	desc = zimbabweElection.65.d

	fire_only_once = yes
	
	trigger = {
		has_country_flag = zim_cw
		has_war = no
		original_tag = ZIM
	}
	mean_time_to_happen = {
	days = 1
	}
	#is_triggered_only = yes
	
	option = {
		name = zimbabweElection.65.a
		clr_country_flag = zim_cw
		if = {
			limit = {
				owns_state = 545			
			}
			set_capital = 545
		}
		if = {
			limit = {
				has_country_flag = rhodesia
			}
			set_province_name = { id = 10929 name = "Salisbury"}
		}
		if = {
			limit = {
				AND = {
					has_country_flag = zimbabwe
					has_government = nationalism
				}
			}
			set_province_name = { id = 10929 name = "Harare"}
		}
		if = {
			limit = {
				has_government = monarchism
			}
			set_province_name = { id = 10929 name = "Great Zimbabwe"}
		}
		
	}
}
