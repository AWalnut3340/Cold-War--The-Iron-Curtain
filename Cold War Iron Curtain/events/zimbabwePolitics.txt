#Event: Leftwing Parties
country_event = {

	id = zimbabwePolitics.1
	title = zimbabwePolitics.1.t
	desc = zimbabwePolitics.1.d
	picture = GFX_report_event_german_speech

	fire_only_once = no

	is_triggered_only = yes
	#Socialism
	option = {
		name = zimbabwePolitics.1.a
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { has_government = socialist }
			}
		}
		set_politics = { ruling_party = socialist elections_allowed = no }
	}
	#Market Socialism
	option = {
		name = zimbabwePolitics.1.b
		set_politics = { ruling_party = Social_Democratic elections_allowed = no }
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { has_government = Social_Democratic }
			}
		}
	}
	#Social Liberalism

		ai_chance = {
			factor = 1
