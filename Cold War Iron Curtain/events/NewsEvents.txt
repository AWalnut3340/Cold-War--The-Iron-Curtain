# End of the Nuremberg trials
news_event = {
	id = news.1
	title = news.1.t
	desc = news.1.d
	picture = GFX_news_event_nuremberg

	major = yes

	immediate = {
		set_global_flag = nuremberg_trials_over
	}

	trigger = {
		date < 1949.5.26
		NOT = { has_global_flag = nuremberg_trials_over }
	}

	option = {
		name = news.1.a
	}
}

# The (first) Korean War breaks out
news_event = {
	id = news.3
	title = news.3.t
	desc = news.3.desc
	picture = GFX_news_event_korea_war
	
	major = yes
	
	immediate = {
		set_global_flag = first_korean_war
	}
	
	trigger = {
		tag = KPA
		has_war_with = KOR
		NOT = { has_global_flag = first_korean_war}
	}
	
	option = {
		name = news.3.a
		trigger = {
			tag = USA
		}
	}
		
	option = {
		name = news.3.b
		trigger = {
			tag = SOV
		}
	}
	
	option = {
		name = news.3.c
		trigger = {
			NOT = {
				tag = USA
				tag = SOV
				tag = KOR
				tag = KPA
			}
		}
	}
	
	option = {
		name = news.3.d
		trigger = {
			tag = KPA
		}
	}
	
	option = {
		name = news.3.e
		trigger = {
			tag = KOR
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
}
